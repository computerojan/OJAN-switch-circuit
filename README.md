# OJAN-switch-circuit
Repository for ojan switch circuit 

#include<stdio.h>    
int main(){ 
    printf("WELCOME TO SWITCH CIRCUIT");  
    
int number;    
printf("ENTER NUMBER TO SWITCH ON");    
scanf("%d",&number);     
if(number==1){    
printf("SWITCH IS ON");    
}    
else if(number==0){    
printf("SWITCH IS OFF");    
}    
else if(number==2){    
printf("SWITCH IS OFF");    
}    
else{    
printf("INVALID CHOICE! PLEASE TRY AGAIN");    
}    
return 0;  
}
